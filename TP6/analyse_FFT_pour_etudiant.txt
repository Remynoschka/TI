		{
		max_1 =p;

		{
		max_2 =p;
		i_max_2 = i;
		j_max_2 =j;
		} 
      	}
   }
  print("Px=",i_max_2);
print("Py=",j_max_2);  
  D=sqrt(pow(i_max_1 - i_max_2,2)+pow(j_max_1 - j_max_2,2));

  if(i_max_1!=i_max_2 && j_max_1!=j_max_2){
	print("texture diagonale");
  }
  else if(i_max_1==i_max_2 && j_max_1!=j_max_2){
	print("texture verticale");
  }
   else if(i_max_1!=i_max_2 && j_max_1==j_max_2){
	print("texture horizontal");
  }
//Q8
  omega=D/sqrt(W*H);
  print("omega=",omega);
}
